def parse_fasta(data):
    """Parse the input data in FASTA format and return a dictionary of string ID and sequence."""
    sequences = {}
    current_id = None
    current_seq = []  # To store the sequence data for each ID
    
    # Split the input string into lines
    lines = data.splitlines()
    
    for line in lines:
        line = line.strip()
        if line.startswith(">"):
            # If we were processing a sequence, store it
            if current_id is not None:
                sequences[current_id] = ''.join(current_seq)
            # Now handle the new sequence record
            current_id = line[1:]  # Remove the '>' and set the current ID
            current_seq = []  # Reset the current sequence
        else:
            current_seq.append(line)  # Collect sequence lines

    # Don't forget to store the last sequence
    if current_id is not None:
        sequences[current_id] = ''.join(current_seq)

    return sequences

def build_overlap_graph(sequences, k):
    """Build the overlap graph for the given sequences with overlap length k."""
    overlap_graph = {}
    
    # Iterate through each pair of strings to check for overlap
    for s_id, s_seq in sequences.items():
        overlap_graph[s_id] = []  # Initialize the adjacency list for the current string
        suffix_s = s_seq[-k:]  # Last k characters of s
        for t_id, t_seq in sequences.items():
            if s_id != t_id:  # Avoid self-loops
                prefix_t = t_seq[:k]  # First k characters of t
                if suffix_s == prefix_t:  # Check if they overlap
                    overlap_graph[s_id].append(t_id)
    
    return overlap_graph

def print_adjacency_list(overlap_graph):
    """Print the adjacency list of the graph."""
    for node, neighbors in overlap_graph.items():
        for neighbor in neighbors:
            print(f"{node} {neighbor}")
